Configure Apps Script URL

To enable Add / Edit / Delete to auto-save into your Google Sheet, deploy an Apps Script Web App from your spreadsheet and paste the URL below.

Steps: Open sheet → Extensions → Apps Script → paste the provided script → Deploy → Web App → copy URL

📋 Copy the Apps Script code to deploy 
function doGet(e) {
  try {
    if (!e.parameter.data) return buildRes({status:"ok",message:"live"});
    var data = JSON.parse(e.parameter.data);
    var ss = SpreadsheetApp.openById("1wkPRTDbL17zt9FJABID8leH3O2-tvwky");
    var sheet = ss.getSheetByName(data.tab);
    if (!sheet) throw new Error("Sheet not found: " + data.tab);
    var headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0];
    if (data.action === "append") {
      sheet.appendRow(headers.map(function(h){ return data.row[h]||""; }));
    } else if (data.action === "update" || data.action === "delete") {
      var rows = sheet.getDataRange().getValues();
      var keySearch = String(data.key).split("|")[0].trim();
      for (var i=1; i<rows.length; i++) {
        if (String(rows[i][0]).trim() === keySearch) {
          if (data.action === "update") {
            headers.forEach(function(h,j){ if(data.row[h]!==undefined) sheet.getRange(i+1,j+1).setValue(data.row[h]); });
          } else { sheet.deleteRow(i+1); }
          break;
        }
      }
    }
    return buildRes({status:"ok"});
  } catch(err) { return buildRes({status:"error",message:err.message}); }
}
function doPost(e) { return doGet(e); }
function buildRes(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
